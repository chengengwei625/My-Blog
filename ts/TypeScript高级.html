partial(转化可选) #

作用：将传入属性变为可选项

可把定义好的对象（包含 必选+可选项）类型全部转化为可选项

源码：

type Partial<T> = {
    [P in keyof T]?: T[P]
}
1
2
3

用例:

 // 已有定义类型Person
 interface Person {
     name: string;
     age: number;
     id: number;
     sex: 0 | 1;
     address: string;
     weight: number;
 }
 
 // 使用方法
 const newObj: Partial<Person> = {
     name: '张三' // 假如只需要一项 Partial的便捷性 可以不需要从新定义类型
 };
 
 // Partial<Person>等同于 NewPerson
 interface NewPerson {
     name?: string;
     age?: number;
     id?: number;
     sex?: 0 | 1;
     address?: string;
     weight?: number;
 }

Required(转化必选) #

作用：将传入的属性变为必选项

Required 和 Partial刚好相反,可把定义好的对象（包含 必选+可选项）类型全部转化为 必选项

源码：

type Required<T> = { [P in keyof T]-?: T[P] };
1

用例:

 // 已有定义类型Person
 interface Person {
     name: string;
     age: number;
     id?: number;
     sex?: 0 | 1;
 }
 
 // 使用方法
 const newObj: Required<Person> = {
     name: '张三',
     age: 1,
     id: 1,
     sex: 1
 };
 
 // Required<Person>等同于 NewPerson
 interface NewPerson {
     name: string;
     age: number;
     id: number;
     sex: 0 | 1;
 }

Pick(取出) #

作用：取出一系列k的属性

源码：

type Pick<T, K extends keyof T> = {    [P in K]: T[P]; };
1

用例:

 interface UserObj {
     readonly name: string;
     age: number;
     id: number;
     sex: 0 | 1;
     address: string;
     weight: number;
  }
  
  // 采集需要的
  type Person = Pick<UserObj, "name" | "id">;
  
  // 此时Person 等同于 Person1
  interface Person1 {
      readonly name: string;
      id: number;
 }

Exclude(排除) #

作用：如果 T 是 U 的子类型则返回 never 不是则返回 T

源码:

type Exclude<T, U> = T extends U ? never : T
1

用例:

 type A = number | string | boolean
 type B = number | boolean
 
 type Foo = Exclude<A, B>
 // 相当于
 type Foo = string

Extract（提取/包括） #

作用：和 Exclude 相反

源码：

type Exclude<T, U> = T extends U ? never : T;
1

用例:

 type A = number | string | boolean
 type B = number | boolean
 
 type Foo = Extract<A, B>
 // 相当于
 type Foo = number | boolean
 type test = Extract<'a' | 'b' | 'c' | 'd', 'a' | 'c' | 'f'|'g'>;  // -> 'a' | 'c'
 // Extract首先是取出的意思，应用Extract条件类型，会尝试寻找T类型成员和U类型成员的交集，在该示例中，取到的是"a" | "c"

Omit(剔除) #

作用：把后者存在的属性，从前者中剔除

源码:

type Omit<T, K extends keyof any> = Pick<T, Exclude<keyof T, K>>;
1

用例：

 interface UserObj {
     readonly name: string; // readonly 只读属性 只能初始化定义 不能二次赋值
     age: number;
     id: number;
     sex: 0 | 1;
     address: string;
     weight: number;
 }
 
 // 剔除省略自己不需要的
 type Person = Omit<UserObj , "number" | "sex"  | "address" | "weight">;
 
 // 此时Person 等同于 Person1
 
 interface Person1 {
     readonly name: string;
     id: number;
 }

Record(记录) #

作用：以 typeof 格式快速创建一个类型，此类型包含一组指定的属性且都是必填

用例：

 type Coord = Record<'x' | 'y', number>;
 
 // 等同于
 type Coord = {
 	x: number;
 	y: number;
 }

Readonly(转化只读) #

作用:Readonly 就是为类型对象每一项添加前缀 Readonly

源码:

type Readonly<T> = {    readonly [P in keyof T]: T[P]; };
1

用例:

interface Person {
    readonly name: string; // 只有这一项有readonly
    age: number;
    id?: number;
}

// 使用方法
const newObj: Readonly<Person> = {
    name: '张三',
    age: 1,
    id: 1
};
// newObj.name = '李四'; // 异常 因为有readonly只读属性，只能初始化声明，不能赋值。

// Readonly<Person> 等同于 NewPerson
interface NewPerson {
    readonly name: string;
    readonly age: number;
    readonly id?: number;
}
